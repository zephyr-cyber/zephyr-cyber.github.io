# RSArepeat

这题好像也有层一样的，比赛真 lao
给出加密脚本

#!/usr/bin/env python
# -*- coding: utf-8 -*-
from Crypto.Util.number import *
import gmpy2
import os
from secret import flag


p = gmpy2.next_prime(bytes_to_long(os.urandom(32) * 10))
q = getPrime(2048)
n = p * q
m = bytes_to_long(flag)
e = 65537
c = pow(m, e, n)
print 'n = ' + str(n)
print 'c = ' + str(c)
